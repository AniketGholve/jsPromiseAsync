        Q7 Create examples to explain promises function

        function promiseFunction(val){
            return new Promise((res,rej)=>{
                 if(val!=null){
                     res("Promise Resolved");
                 }
                 else{
                     rej("Promise Rejected")
                 }
             })
         }
         
         promiseFunction("Aniket").then((res)=>console.log(res),(err)=>{console.log(err)}) // Promise Resolved
         promiseFunction(null).then((res)=>console.log(res),(err)=>{console.log(err)}) // Promise Rejected

        the function will return the promise if value passed is null then it will reject the promise and if value is not null the it will accept the promise

        Q9 Create examples to explain promise.all function

        async function promiseFunction1(){
            return new Promise((res)=>{
                setTimeout(()=>{
                    res("Resolved Data1");
                },2000)
            })
        }
        async function promiseFunction2(){
            return new Promise((res)=>{
                setTimeout(()=>{
                    res("Resolved Data2");
                },3000)
            })
        }
        async function promiseFunction3(){
            return new Promise((res)=>{
                setTimeout(()=>{
                    res("Resolved Data3");
                },4000)
            })
        }

        console.log("Start")
        let [data1,data2,data3]=await Promise.all([
            promiseFunction1(),
            promiseFunction2(),
            promiseFunction3()
        ]);
        console.log(data1,data2,data3);
        console.log("End")
        output //
        Start
        Resolved Data1 Resolved Data2 Resolved Data3
        End
        
        Promise all will wait for all the data to process and send all the data at a time
